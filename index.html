function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Daily Summary'); var data = JSON.parse(e.postData.contents); var incomingDate = normaliseDate(data.date); ensureAppHeaders(sheet); var targetRow = findRowByDate(sheet, incomingDate); if (!targetRow) { sheet.appendRow([incomingDate]); targetRow = sheet.getLastRow(); } sheet.getRange(targetRow, 48).setValue(data.bodyweight || ''); sheet.getRange(targetRow, 49).setValue(data.sessionVolume || ''); sheet.getRange(targetRow, 50).setValue(data.hrPeak || ''); sheet.getRange(targetRow, 51).setValue(data.hrAvg || ''); sheet.getRange(targetRow, 52).setValue(data.warmupSkipped ? 'YES' : 'NO'); sheet.getRange(targetRow, 53).setValue(data.exerciseLog || ''); sheet.getRange(targetRow, 54).setValue(data.note || ''); sheet.getRange(targetRow, 55).setValue(data.focusType || ''); sheet.getRange(targetRow, 56).setValue(data.primaryLiftsSkipped ? 'YES' : 'NO'); sheet.getRange(targetRow, 57).setValue(data.strikingVolumeMode || ''); var legacy = sheet.getRange(targetRow, 4, 1, 9).getValues()[0]; sheet.getRange(targetRow, 4, 1, 9).setValues([[ data.workoutType || legacy[0], data.condition || legacy[1], data.sleep || legacy[2], data.duration || legacy[3], data.sectionsDone ? data.sectionsDone.split('/')[0] : legacy[4], data.sectionsDone ? data.sectionsDone.split('/')[1] : legacy[5], data.failureCount !== undefined ? data.failureCount : legacy[6], data.skipCount !== undefined ? data.skipCount : legacy[7], legacy[8] ]]); return ContentService.createTextOutput(JSON.stringify({status:'ok',row:targetRow})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:'error',message:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status:'ok'})).setMimeType(ContentService.MimeType.JSON); } function ensureAppHeaders(sheet) { var h = {48:'Bodyweight',49:'Session_Volume_Lbs',50:'Peak_HR',51:'Avg_HR_Work',52:'Warmup_Skipped',53:'Exercise_Log',54:'Session_Note_App',55:'Focus_Type',56:'Primary_Lifts_Skipped',57:'Striking_Volume_Mode'}; for (var c in h) { var cell = sheet.getRange(1, parseInt(c)); if (!cell.getValue()) cell.setValue(h[c]); } } function findRowByDate(sheet, dateStr) { var last = sheet.getLastRow(); if (last < 2) return null; var dates = sheet.getRange(2, 1, last - 1, 1).getValues(); for (var i = 0; i < dates.length; i++) { var v = dates[i][0]; var s = v instanceof Date ? (v.getMonth()+1)+'/'+v.getDate()+'/'+v.getFullYear() : String(v).trim(); if (s === dateStr) return i + 2; } return null; } function normaliseDate(raw) { if (!raw) return ''; var s = String(raw).trim(); if (/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(s)) return s; var iso = s.match(/^(\d{4})-(\d{2})-(\d{2})/); if (iso) return parseInt(iso[2])+'/'+parseInt(iso[3])+'/'+iso[1]; var m = {Jan:1,Feb:2,Mar:3,Apr:4,May:5,Jun:6,Jul:7,Aug:8,Sep:9,Oct:10,Nov:11,Dec:12}; var f = s.match(/([A-Z][a-z]{2})\s+(\d{1,2}),?\s+(\d{4})/); if (f && m[f[1]]) return m[f[1]]+'/'+parseInt(f[2])+'/'+f[3]; return s; }
